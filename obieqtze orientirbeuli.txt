int upperCase(string word)
{
    int lettersCount{0};
    int a = word.length();
    char array[a + 1];

    strcpy(array, word.c_str());
    for (int i = 0; i < a; i++)
    {
        if (array[i] >= 97 && array[i] <= 97 + 26)
        {
            lettersCount++;
        }
    }
    return lettersCount;
}
